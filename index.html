  <script type="text/babel">

    const { useState, useRef, useEffect } = React;


    const CHANNEL_URL = "https://www.youtube.com/@85collective";


    // ── Replace with your real YouTube video IDs ──────────────────

    const EPISODES = [

      { id: "VIDEO_ID_4", num: "004", title: "Radio Show #004", artists: ["Artist F"],             date: "Apr 2025", dur: "55:44"  },

      { id: "VIDEO_ID_5", num: "005", title: "Radio Show #005", artists: ["Artist G", "Artist H"], date: "May 2025", dur: "1:04:20" },

      { id: "VIDEO_ID_6", num: "006", title: "Radio Show #006", artists: ["Artist I"],             date: "Jun 2025", dur: "1:01:07" },

    ];

    // ─────────────────────────────────────────────────────────────


    const ytThumb = id => `https://img.youtube.com/vi/${id}/maxresdefault.jpg`;


    function Thumb({ id, alt }) {

      const [err, setErr] = useState(false);

      const s = { width:"100%", height:"100%", display:"block" };

      return err

        ? <div style={{ ...s, background:"linear-gradient(145deg,#181818,#2e2e2e)" }} />

        : <img src={ytThumb(id)} alt={alt||""} onError={() => setErr(true)} style={{ ...s, objectFit:"cover" }} />;

    }


    function EpCard({ ep, onWatch, onAudio, isActive, isPlaying }) {

      return (

        <div className="card">

          <div style={{ aspectRatio:"16/9", overflow:"hidden", background:"#f0f0f0", marginBottom:16, position:"relative" }}>

            <div className="ti" style={{ width:"100%", height:"100%" }}>

              <Thumb id={ep.id} alt={ep.title} />

            </div>

            <div className="ov" style={{ position:"absolute", inset:0, background:"rgba(0,0,0,.55)", display:"flex", alignItems:"center", justifyContent:"center", gap:10 }}>

              <button className="wbtn" onClick={() => onWatch(ep)}

                style={{ color:"#fff", fontSize:11, fontWeight:700, letterSpacing:"0.1em", textTransform:"uppercase", padding:"8px 16px", border:"1px solid rgba(255,255,255,.35)", background:"transparent" }}>

                Watch

              </button>

              <button className="wbtn" onClick={() => onAudio(ep)}

                style={{ color:"#fff", fontSize:11, fontWeight:700, letterSpacing:"0.1em", textTransform:"uppercase", padding:"8px 16px", border:"1px solid rgba(255,255,255,.35)", background:"transparent" }}>

                {isPlaying ? "⏸" : "♪ Audio"}

              </button>

            </div>

          </div>

          <div style={{ display:"flex", alignItems:"baseline", gap:10, marginBottom:5 }}>

            <span style={{ fontSize:9, fontWeight:700, color:"#ccc", letterSpacing:"0.12em", textTransform:"uppercase", flexShrink:0 }}>EP {ep.num}</span>

            <span style={{ fontSize:15, fontWeight:700, lineHeight:1.3 }}>{ep.title}</span>

          </div>

          <p style={{ fontSize:13, color:"#666", marginBottom:3 }}>{ep.artists.join(", ")}</p>

          <p style={{ fontSize:11, color:"#bbb" }}>{ep.date} · {ep.dur}</p>

          <div style={{ display:"flex", gap:14, marginTop:12 }}>

            <button onClick={() => onWatch(ep)} style={{ fontSize:11, fontWeight:700, color:"#888", letterSpacing:"0.06em", textTransform:"uppercase", borderBottom:"1px solid #ddd", paddingBottom:1 }}>Watch</button>

            <button onClick={() => onAudio(ep)} style={{ fontSize:11, fontWeight:700, letterSpacing:"0.06em", textTransform:"uppercase", paddingBottom:1, borderBottom:`1px solid ${isActive?"#111":"#ddd"}`, color:isActive?"#111":"#888" }}>

              {isPlaying ? "⏸ Pause" : "♪ Listen"}

            </button>

          </div>

        </div>

      );

    }

    function App() {

      const [view,    setView]    = useState("episodes");

      const [watchEp, setWatchEp] = useState(null);

      const [audioEp, setAudioEp] = useState(null);

      const [playing, setPlaying] = useState(false);

      const [ready,   setReady]   = useState(false);

      const iframeRef = useRef(null);


      useEffect(() => {

        if (!ready) return;

        try {

          iframeRef.current?.contentWindow?.postMessage(

            JSON.stringify({ event:"command", func: playing ? "playVideo":"pauseVideo", args:[] }), "*"

          );

        } catch(_) {}

      }, [playing, ready]);


      const handleAudio = ep => {

        if (audioEp?.id === ep.id) { setPlaying(p => !p); }

        else { setReady(false); setAudioEp(ep); setPlaying(true); }

      };

      const stopAudio = () => { setAudioEp(null); setPlaying(false); setReady(false); };


      const latest  = EPISODES[EPISODES.length - 1];

      const allDesc = [...EPISODES].reverse();


      return (

        <div style={{ minHeight:"100vh", background:"#fff", color:"#111", paddingBottom: audioEp ? 76 : 0 }}>


          {/* HEADER */}

          <header style={{ position:"sticky", top:0, zIndex:100, background:"rgba(255,255,255,.96)", backdropFilter:"blur(12px)", borderBottom:"1px solid #ebebeb", height:64, padding:"0 40px", display:"flex", alignItems:"center", justifyContent:"space-between" }}>

            <button onClick={() => setView("episodes")} style={{ fontWeight:800, fontSize:13, letterSpacing:"0.22em", textTransform:"uppercase" }}>

              85 Collective

            </button>

            <nav style={{ display:"flex", gap:28, fontSize:11, fontWeight:700, letterSpacing:"0.12em", textTransform:"uppercase", alignItems:"center" }}>

              {["episodes","about"].map(v => (

                <button key={v} className="nb" onClick={() => setView(v)}

                  style={{ color:view===v?"#111":"#bbb", paddingBottom:2, borderBottom:view===v?"1.5px solid #111":"1.5px solid transparent" }}>

                  {v}

                </button>

              ))}

              <a href={CHANNEL_URL} target="_blank" rel="noopener noreferrer" className="nb" style={{ color:"#bbb" }}>YouTube ↗</a>

            </nav>

          </header>


          {/* MAIN */}

          <main style={{ maxWidth:1080, margin:"0 auto", padding:"60px 40px" }}>

            {view === "episodes" && <>

              {/* Hero */}

              <section style={{ marginBottom:80 }}>

                <p style={{ fontSize:10, fontWeight:700, letterSpacing:"0.22em", textTransform:"uppercase", color:"#bbb", marginBottom:24 }}>Latest Episode</p>

                <div style={{ display:"grid", gridTemplateColumns:"1.15fr 0.85fr", gap:56, alignItems:"center" }}>

                  <div style={{ aspectRatio:"16/9", overflow:"hidden", background:"#f0f0f0" }}>

                    <Thumb id={latest.id} alt={latest.title} />

                  </div>

                  <div>

                    <div style={{ fontSize:108, fontWeight:900, color:"#efefef", lineHeight:.85, marginBottom:20, letterSpacing:"-0.05em", userSelect:"none" }}>{latest.num}</div>

                    <h1 style={{ fontSize:26, fontWeight:800, lineHeight:1.2, marginBottom:10 }}>{latest.title}</h1>

                    <p style={{ fontSize:14, color:"#555", marginBottom:6 }}>{latest.artists.join(", ")}</p>

                    <p style={{ fontSize:12, color:"#bbb", marginBottom:36 }}>{latest.date} · {latest.dur}</p>

                    <div style={{ display:"flex", gap:10 }}>

                      <button className="btn-w" onClick={() => setWatchEp(latest)}

                        style={{ padding:"11px 26px", background:"#111", color:"#fff", fontSize:11, fontWeight:700, letterSpacing:"0.1em", textTransform:"uppercase" }}>

                        Watch

                      </button>

                      <button className="btn-l" onClick={() => handleAudio(latest)}

                        style={{ padding:"11px 26px", border:"1px solid #ddd", fontSize:11, fontWeight:700, letterSpacing:"0.1em", textTransform:"uppercase" }}>

                        {audioEp?.id === latest.id && playing ? "⏸ Pause" : "♪ Listen"}

                      </button>

                    </div>

                  </div>

                </div>

              </section>


              <div style={{ borderTop:"1px solid #ebebeb", marginBottom:56 }} />


              <p style={{ fontSize:10, fontWeight:700, letterSpacing:"0.22em", textTransform:"uppercase", color:"#bbb", marginBottom:36 }}>All Episodes</p>

              <div style={{ display:"grid", gridTemplateColumns:"repeat(3,1fr)", gap:"52px 28px" }}>

                {allDesc.map(ep => (

                  <EpCard key={ep.id} ep={ep} onWatch={setWatchEp} onAudio={handleAudio}

                    isActive={audioEp?.id === ep.id} isPlaying={audioEp?.id === ep.id && playing} />

                ))}

              </div>

            </>}

            {view === "about" && <About />}

          </main>


          {/* WATCH MODAL */}

          {watchEp && (

            <div onClick={() => setWatchEp(null)} style={{ position:"fixed", inset:0, background:"rgba(0,0,0,.93)", zIndex:300, display:"flex", alignItems:"center", justifyContent:"center" }}>

              <div onClick={e => e.stopPropagation()} style={{ width:"min(92vw,960px)", aspectRatio:"16/9" }}>

                <iframe src={`https://www.youtube.com/embed/${watchEp.id}?autoplay=1`}

                  style={{ width:"100%", height:"100%", border:"none" }}

                  allow="autoplay;fullscreen" allowFullScreen />

              </div>

              <button onClick={() => setWatchEp(null)} style={{ position:"absolute", top:18, right:24, color:"#fff", fontSize:36, opacity:.5, lineHeight:1, fontWeight:300 }}>×</button>

            </div>

          )}


          {/* AUDIO PLAYER */}

          {audioEp && <>

            <iframe key={audioEp.id} ref={iframeRef}

              src={`https://www.youtube.com/embed/${audioEp.id}?enablejsapi=1&autoplay=1&controls=0`}

              onLoad={() => setReady(true)}

              style={{ position:"fixed", top:-9999, left:0, width:1, height:1 }}

              allow="autoplay" title="audio" />

            <div style={{ position:"fixed", bottom:0, left:0, right:0, height:76, background:"#111", color:"#fff", display:"flex", alignItems:"center", gap:20, padding:"0 40px", zIndex:200, borderTop:"1px solid #1e1e1e" }}>

              <div style={{ width:44, height:44, flexShrink:0, overflow:"hidden", background:"#222" }}>

                <Thumb id={audioEp.id} alt="" />

              </div>

              <div style={{ flex:1, minWidth:0 }}>

                <p style={{ fontSize:9, color:"#555", textTransform:"uppercase", letterSpacing:"0.16em", marginBottom:3 }}>Audio Only</p>

                <p style={{ fontSize:13, fontWeight:700, whiteSpace:"nowrap", overflow:"hidden", textOverflow:"ellipsis" }}>{audioEp.title}</p>

                <p style={{ fontSize:11, color:"#666" }}>{audioEp.artists.join(", ")}</p>

              </div>

              <button onClick={() => setPlaying(p => !p)}

                style={{ width:44, height:44, borderRadius:"50%", border:"1px solid #333", color:"#fff", display:"flex", alignItems:"center", justifyContent:"center", fontSize:15, flexShrink:0 }}>

                {playing ? "⏸" : "▶"}

              </button>

              <button onClick={stopAudio} style={{ color:"#444", fontSize:28, fontWeight:300, lineHeight:1, flexShrink:0 }}>×</button>

            </div>

          </>}

        </div>

      );

    }


    const root = ReactDOM.createRoot(document.getElementById("root"));

    root.render(<App />);

  </script>
